pipeline {
    agent any
    tools{
        jdk 'jdkv'
        maven 'mavenv'
    }
    
    environment {
        scanner = tool 'sonarqube'
        IMAGE_TAG = "v${BUILD_NUMBER}"

    }

    stages {
        stage('Git checkout') {
            steps {
                git branch: 'main', changelog: false, credentialsId: 'git-cred', poll: false, url: 'https://github.com/Hadeedahmed254/CICD-3-CI'
            }
        }
        
        stage('Check the code is Syntaxically coorrect') {
            steps {
                sh "mvn compile"
            }
        }
        
        stage('check the code is logically correct') {
            steps {
                sh  "mvn test"
            }
        }
        
        stage('scan to check vulnerablities in dependencies') {
            steps {
                sh "trivy fs --format table -o trivy-fs-index.html ."
            }
        }
        
        stage('sonarqube scanner analysis') {
            steps {
                withSonarQubeEnv('sonar') {
                        sh ''' $scanner/bin/sonar-scanner -Dsonar.projectName==multitierapp -Dsonar.projectKey=multitierap \
                        -Dsonar.java.binaries=. '''
                }
            }
        }
        
        stage('Peform quality gate by Sonar Server') {
            steps {
                script {
                    waitForQualityGate abortPipeline: false, credentialsId: 'sonnar-cred'
                }
            }
        }
        
        stage('build the soruce code and resources into .jar') {
            steps {
                sh "mvn package"
            }
        }
        
        stage('publish artifacts to nexus') {
            steps {
                withMaven(globalMavenSettingsConfig: 'nexus-id', jdk: 'jdkv', maven: 'mavenv', traceability: true) {
                            sh "mvn deploy"
                }
            }
        }
        
        stage('build and tag docker image') {
            steps {
                script {
                    withDockerRegistry(credentialsId: 'docker-cred') {
                        sh "docker build -t hadeedahmed254/cicd:${IMAGE_TAG} ."
                    }
                }
            }
        }
        
        
        
        stage('scan docker image') {
            steps {
                sh "trivy image --format table -o trivy-image-index.html hadeedahmed254/cicd:${IMAGE_TAG}"
            }
        }
        
        stage('Push docker image') {
            steps {
                script {
                    withDockerRegistry(credentialsId: 'docker-cred') {
                        sh "docker push hadeedahmed254/cicd:${IMAGE_TAG}"
                    }
                }
            }
        }
        stage('Update menifest in CD repo') {
            steps {
                script {
            // Clean workspace before starting
            cleanWs()

            withCredentials([usernamePassword(credentialsId: 'git-cred', usernameVariable: 'GIT_USERNAME', passwordVariable: 'GIT_PASSWORD')]) {
                sh ''' # Clone the Mega-Project-CD repository
                git clone https://${GIT_USERNAME}:${GIT_PASSWORD}@github.com/Hadeedahmed254/CICD-3-CD

                # Update the image tag in the manifest.yaml file
                cd CICD-3-CD
                sed -i "s|hadeedahmed254/cicd:.*|hadeedahmed254/cicd:${IMAGE_TAG}|" menifest.yaml

                # Confirm changes
                echo "Updated manifest file contents:"
                cat menifest.yaml

                # Commit and push the changes
                git config user.name "Jenkins"
                git config user.email "jenkins@example.com"
                git add menifest.yaml
                git commit -m "Update image tag to ${IMAGE_TAG}"
                git push origin main
                '''
                        }
                    }
                }
            }

        }
    post {
    always {
        script {
            def jobName = env.JOB_NAME
            def buildNumber = env.BUILD_NUMBER
            def pipelineStatus = currentBuild.result ?: 'UNKNOWN'
            def bannerColor = pipelineStatus.toUpperCase() == 'CI pipeline succesfully deployed' ? 'green' : 'red'

            def body = """
                <html>
                <body>
                <div style="border: 4px solid ${bannerColor}; padding: 10px;">
                <h2>${jobName} - Build ${buildNumber}</h2>
                <div style="background-color: ${bannerColor}; padding: 10px;">
                <h3 style="color: white;">Pipeline Status: ${pipelineStatus.toUpperCase()}</h3>
                </div>
                <p>Check the <a href="${BUILD_URL}">console output</a>.</p>
                </div>
                </body>
                </html>
            """

            emailext (
                subject: "${jobName} - Build ${buildNumber} - ${pipelineStatus.toUpperCase()}",
                body: body,
                to: 'hadeeda5@gmail.com',
                from: 'jenkins@example.com',
                replyTo: 'jenkins@example.com',
                mimeType: 'text/html',
                attachmentsPattern: 'trivy-image-index.html'
            )
        }
    }
    
  }    

}
